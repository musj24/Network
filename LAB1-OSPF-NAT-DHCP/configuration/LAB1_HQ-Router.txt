
# [0] 기본 라우팅 (디폴트 라우트)
ip route 0.0.0.0 0.0.0.0 203.0.113.2

# [1] 인터페이스 설정
! --- HQ 내부 LAN (192.168.1.0/24) ---
interface g0/1
 ip address 192.168.1.1 255.255.255.0
 ip helper-address 192.168.4.2
 ip nat inside

! --- HQ <-> Branch 전용 링크 (192.168.3.0/30) ---
interface g0/0
 ip address 192.168.3.1 255.255.255.252
 ip nat inside

! --- HQ <-> ISP 외부 링크 (203.0.113.0/30) ---
interface g0/2
 ip address 203.0.113.1 255.255.255.252
 ip nat outside
 ip nat inside source list 1 int g0/2 overload
 ip nat inside source static 192.168.5.2 203.1.113.1

# [2] 라우팅 프로토콜 (OSPF)
router ospf 1
 router-id 1.1.1.1
 network 192.168.1.0 0.0.0.255 area 0
 network 192.168.3.0 0.0.0.3 area 0
 network 203.0.113.0 0.0.0.3 area 0
 default-information originate

# [3] NAT 대상 트래픽 매칭(ACL)
access-list 1 permit 192.168.1.0 0.0.0.255
access-list 1 permit 192.168.2.0 0.0.0.255
access-list 1 permit 192.168.5.0 0.0.0.3
